     1→4.3. Rate Limiter (Scraping Koruma)
     2→typescript
     3→// apps/scraper/src/utils/rateLimiter.ts
     4→
     5→interface RateLimiterConfig {
     6→  maxRequests: number;
     7→  perSeconds: number;
     8→  key: string;
     9→}
    10→
    11→export class RateLimiter {
    12→  private maxRequests: number;
    13→  private perSeconds: number;
    14→  private key: string;
    15→  private requests: number[] = [];
    16→  private lock: Promise<void> = Promise.resolve();
    17→
    18→  constructor(config: RateLimiterConfig) {
    19→    this.maxRequests = config.maxRequests;
    20→    this.perSeconds = config.perSeconds;
    21→    this.key = config.key;
    22→  }
    23→
    24→  async wait(): Promise<void> {
    25→    await this.lock;
    26→    
    27→    this.lock = this.lock.then(async () => {
    28→      const now = Date.now();
    29→      const windowStart = now - (this.perSeconds * 1000);
    30→      
    31→      // Eski istekleri temizle
    32→      this.requests = this.requests.filter(time => time > windowStart);
    33→      
    34→      if (this.requests.length >= this.maxRequests) {
    35→        const oldestRequest = this.requests[0];
    36→        const waitTime = (oldestRequest + (this.perSeconds * 1000)) - now;
    37→        
    38→        if (waitTime > 0) {
    39→          console.log(`[RateLimiter:${this.key}] Waiting ${waitTime}ms`);
    40→          await new Promise(resolve => setTimeout(resolve, waitTime + 100));
    41→        }
    42→      }
    43→      
    44→      this.requests.push(Date.now());
    45→    });
    46→    
    47→    await this.lock;
    48→  }
    49→
    50→  getStats(): { requests: number; maxRequests: number; windowSeconds: number } {
    51→    const now = Date.now();
    52→    const windowStart = now - (this.perSeconds * 1000);
    53→    const activeRequests = this.requests.filter(time => time > windowStart);
    54→    
    55→    return {
    56→      requests: activeRequests.length,
    57→      maxRequests: this.maxRequests,
    58→      windowSeconds: this.perSeconds
    59→    };
    60→  }
    61→}

   263→6. DEDUPLICATOR (Çapraz-Kaynak Tekilleştirme)
   264→typescript
   265→// apps/scraper/src/deduplicator.ts
   266→
   267→import { PrismaClient } from '@prisma/client';
   268→import { ListingRaw } from './adapters/base';
   269→
   270→const prisma = new PrismaClient();
   271→
   272→export class Deduplicator {
   273→  // Gruplama stratejileri
   274→  private static readonly MAX_PRICE_DIFF_PERCENT = 0.05; // %5
   275→  private static readonly MAX_MILEAGE_DIFF_PERCENT = 0.05; // %5
   276→  private static readonly SAME_YEAR_ALLOWED = 1; // ±1 yıl
   277→
   278→  static async groupDuplicates(listings: ListingRaw[]): Promise<Map<string, string[]>> {
   279→    const groups = new Map<string, string[]>();
   280→    const processed = new Set<string>();
   281→
   282→    for (const listing of listings) {
   283→      if (processed.has(listing.sourceUrl)) continue;
   284→      
   285→      const groupKey = await this.findOrCreateGroup(listing);
   286→      if (!groupKey) continue;
   287→      
   288→      if (!groups.has(groupKey)) {
   289→        groups.set(groupKey, []);
   290→      }
   291→      
   292→      groups.get(groupKey)!.push(listing.sourceUrl);
   293→      processed.add(listing.sourceUrl);
   294→    }
   295→
   296→    return groups;
   297→  }
   298→
   299→  private static async findOrCreateGroup(listing: ListingRaw): Promise<string | null> {
   300→    // 1. VIN ile eşleştirme (en güvenilir)
   301→    if (listing.vin) {
   302→      const existing = await prisma.listing.findFirst({
   303→        where: {
   304→          vin: listing.vin,
   305→          isActive: true,
   306→          id: { not: listing.id }
   307→        }
   308→      });
   309→      
   310→      if (existing) {
   311→        return existing.duplicateGroupId || `vin:${listing.vin}`;
   312→      }
   313→    }
   314→
   315→    // 2. Make+Model+Year + Benzer Km + Benzer Fiyat + Aynı Şehir
   316→    const similar = await prisma.listing.findMany({
   317→      where: {
   318→        make: listing.make,
   319→        model: listing.model,
   320→        year: {
   321→          gte: listing.year - this.SAME_YEAR_ALLOWED,
   322→          lte: listing.year + this.SAME_YEAR_ALLOWED
   323→        },
   324→        city: listing.city,
   325→        isActive: true,
   326→        id: { not: listing.id }
   327→      },
   328→      select: {
   329→        id: true,
   330→        price: true,
   331→        mileageKm: true,
   332→        duplicateGroupId: true
   333→      }
   334→    });
   335→
   336→    for (const candidate of similar) {
   337→      // Km farkı kontrolü
   338→      const kmDiff = Math.abs((candidate.mileageKm || 0) - (listing.mileageKm || 0));
   339→      const kmPercent = kmDiff / Math.max(candidate.mileageKm || 1, listing.mileageKm || 1);
   340→      
   341→      // Fiyat farkı kontrolü
   342→      const priceDiff = Math.abs(candidate.price - listing.price);
   343→      const pricePercent = priceDiff / Math.max(candidate.price, listing.price);
   344→
   345→      if (kmPercent <= this.MAX_MILEAGE_DIFF_PERCENT && 
   346→          pricePercent <= this.MAX_PRICE_DIFF_PERCENT) {
   347→        return candidate.duplicateGroupId || `group:${candidate.id}`;
   348→      }
   349→    }
   350→
   351→    return null;
   352→  }
   353→
   354→  static async getDuplicateGroup(listingId: string): Promise<string[]> {
   355→    const group = await prisma.duplicateGroup.findFirst({
   356→      where: {
   357→        listingIds: {
   358→          has: listingId
   359→        }
   360→      }
   361→    });
   362→    
   363→    return group?.listingIds || [];
   364→  }
   365→}
   366→7. DEĞERLEME MOTORU (Valuator)
   367→typescript
   368→// apps/scraper/src/valuator.ts
   369→
   370→import { PrismaClient } from '@prisma/client';
   371→
   372→const prisma = new PrismaClient();
   373→
   374→interface ValueEstimate {
   375→  value: number;
   376→  confidence: 'high' | 'medium' | 'low' | 'insufficient';
   377→  sampleSize: number;
   378→  minPrice: number;
   379→  maxPrice: number;
   380→  medianPrice: number;
   381→}
   382→
   383→export class Valuator {
   384→  private static readonly MIN_SAMPLE_SIZE = 5;
   385→  private static readonly HIGH_CONFIDENCE_SIZE = 20;
   386→  private static readonly MEDIUM_CONFIDENCE_SIZE = 10;
   387→
   388→  static async updateAllListings(): Promise<void> {
   389→    const listings = await prisma.listing.findMany({
   390→      where: { isActive: true },
   391→      orderBy: { lastSeenAt: 'desc' }
   392→    });
   393→
   394→    let updatedCount = 0;
   395→    for (const listing of listings) {
   396→      try {
   397→        const estimate = await this.estimateValue(listing);
   398→        const dealScore = this.calculateDealScore(listing.price, estimate.value);
   399→        const dealTag = this.getDealTag(dealScore, estimate.confidence);
   400→
   401→        await prisma.listing.update({
   402→          where: { id: listing.id },
   403→          data: {
   404→            estimatedValue: estimate.value,
   405→            confidence: estimate.confidence,
   406→            comparableCount: estimate.sampleSize,
   407→            dealScore: dealScore,
   408→            dealTag: dealTag
   409→          }
   410→        });
   411→        
   412→        updatedCount++;
   413→      } catch (error) {
   414→        console.error(`Error updating listing ${listing.id}:`, error);
   415→      }
   416→    }
   417→
   418→    console.log(`Updated ${updatedCount} listings with valuations`);
   419→  }
   420→
   421→  private static async estimateValue(listing: any): Promise<ValueEstimate> {
   422→    // Comparables: aynı make+model, year ±1, aynı fuel ve transmission
   423→    const comparables = await prisma.listing.findMany({
   424→      where: {
   425→        make: listing.make,
   426→        model: listing.model,
   427→        year: {
   428→          gte: listing.year - 1,
   429→          lte: listing.year + 1
   430→        },
   431→        fuelType: listing.fuelType,
   432→        transmission: listing.transmission,
   433→        isActive: true,
   434→        id: { not: listing.id }
   435→      },
   436→      select: {
   437→        price: true,
   438→        mileageKm: true
   439→      }
   440→    });
   441→
   442→    const sampleSize = comparables.length;
   443→    
   444→    if (sampleSize < this.MIN_SAMPLE_SIZE) {
   445→      return {
   446→        value: listing.price,
   447→        confidence: 'insufficient',
   448→        sampleSize: sampleSize,
   449→        minPrice: listing.price,
   450→        maxPrice: listing.price,
   451→        medianPrice: listing.price
   452→      };
   453→    }
   454→
   455→    // Fiyatları topla ve sırala
   456→    const prices = comparables.map(c => c.price).sort((a, b) => a - b);
   457→    const medianPrice = this.calculateMedian(prices);
   458→    
   459→    // Km düzeltmesi (basit regresyon)
   460→    const mileageAdjustment = this.calculateMileageAdjustment(comparables, listing.mileageKm);
   461→    
   462→    const estimatedValue = Math.round(medianPrice + mileageAdjustment);
   463→    
   464→    // Confidence değerlendirmesi
   465→    let confidence: 'high' | 'medium' | 'low' | 'insufficient';
   466→    if (sampleSize >= this.HIGH_CONFIDENCE_SIZE) {
   467→      confidence = 'high';
   468→    } else if (sampleSize >= this.MEDIUM_CONFIDENCE_SIZE) {
   469→      confidence = 'medium';
   470→    } else {
   471→      confidence = 'low';
   472→    }
   473→
   474→    return {
   475→      value: estimatedValue,
   476→      confidence,
   477→      sampleSize,
   478→      minPrice: prices[0],
   479→      maxPrice: prices[prices.length - 1],
   480→      medianPrice
   481→    };
   482→  }
   483→
   484→  private static calculateMileageAdjustment(comparables: any[], targetMileage: number): number {
   485→    if (comparables.length < 2 || !targetMileage) return 0;
   486→    
   487→    // Basit lineer regresyon: price ~ mileage
   488→    const prices = comparables.map(c => c.price);
   489→    const mileages = comparables.map(c => c.mileageKm || 0);
   490→    
   491→    // Ortalamalar
   492→    const avgPrice = prices.reduce((a, b) => a + b, 0) / prices.length;
   493→    const avgMileage = mileages.reduce((a, b) => a + b, 0) / mileages.length;
   494→    
   495→    // Slope hesapla
   496→    let numerator = 0;
   497→    let denominator = 0;
   498→    for (let i = 0; i < comparables.length; i++) {
   499→      const priceDiff = prices[i] - avgPrice;
   500→      const mileageDiff = (mileages[i] || 0) - avgMileage;
   501→      numerator += priceDiff * mileageDiff;
   502→      denominator += mileageDiff * mileageDiff;
   503→    }
   504→    
   505→    const slope = denominator !== 0 ? numerator / denominator : 0;
   506→    const mileageDiff = (targetMileage || 0) - avgMileage;
   507→    
   508→    return slope * mileageDiff;
   509→  }
   510→
   511→  private static calculateMedian(arr: number[]): number {
   512→    const sorted = [...arr].sort((a, b) => a - b);
   513→    const mid = Math.floor(sorted.length / 2);
   514→    return sorted.length % 2 !== 0 ? sorted[mid] : (sorted[mid - 1] + sorted[mid]) / 2;
   515→  }
   516→
   517→  private static calculateDealScore(price: number, estimatedValue: number): number {
   518→    if (!estimatedValue || estimatedValue === 0) return 0;
   519→    return (price - estimatedValue) / estimatedValue;
   520→  }
   521→
   522→  private static getDealTag(score: number, confidence: string): string {
   523→    if (confidence === 'insufficient') {
   524→      return 'Değerlendirilemedi';
   525→    }
   526→    
   527→    if (score < -0.15) return 'Harika Fırsat';
   528→    if (score < -0.05) return 'İyi Fiyat';
   529→    if (score < 0.05) return 'Piyasa Fiyatı';
   530→    if (score < 0.15) return 'Piyasa Üstü';
   531→    return 'Pahalı';
   532→  }
   533→
   534→  // Belirli bir ilan için manuel değerleme
   535→  static async estimateSingleListing(listingId: string): Promise<ValueEstimate | null> {
   536→    const listing = await prisma.listing.findUnique({
   537→      where: { id: listingId }
   538→    });
   539→    
   540→    if (!listing) return null;
   541→    return this.estimateValue(listing);
   542→  }
   543→}

   600→  private static async calculateOwnershipCost(listing: any): Promise<OwnershipCost> {
   601→    const estimatedValue = listing.estimatedValue || listing.price;
   602→    const age = new Date().getFullYear() - listing.year;
   603→
   604→    // 1. Değer kaybı (amortisman)
   605→    const depreciationRate = this.calculateDepreciationRate(listing.make, listing.model, age);
   606→    const annualDepreciationAmount = estimatedValue * depreciationRate;
   607→    const annualDepreciationPercent = depreciationRate * 100;
   608→
   609→    // 2. Yakıt maliyeti
   610→    const fuelConsumption = this.estimateFuelConsumption(listing.fuelType, listing.bodyType);
   611→    const fuelCostAnnual = (this.ANNUAL_KM / 100) * fuelConsumption * this.FUEL_PRICE_PER_LITER;
   612→
   613→    // 3. Sigorta maliyeti
   614→    const insuranceCostAnnual = estimatedValue * this.INSURANCE_COEFFICIENT;
   615→
   616→    // 4. Bakım maliyeti
   617→    const maintenanceCostAnnual = estimatedValue * this.MAINTENANCE_COEFFICIENT;
   618→
   619→    // 5. Vergi (MTV) - yaklaşık
   620→    const taxCostAnnual = this.calculateTax(listing.year, listing.engineVolume || 0);
   621→
   622→    const totalAnnual = fuelCostAnnual + insuranceCostAnnual + maintenanceCostAnnual + taxCostAnnual;
   623→
   624→    return {
   625→      annualDepreciationPercent,
   626→      annualDepreciationAmount,
   627→      fuelCostAnnual,
   628→      insuranceCostAnnual,
   629→      maintenanceCostAnnual,
   630→      taxCostAnnual,
   631→      totalAnnual: Math.round(totalAnnual)
   632→    };
   633→  }

   635→  private static calculateDepreciationRate(make: string, model: string, age: number): number {
   636→    // Her marka/model için farklı amortisman oranları
   637→    const luxuryBrands = ['bmw', 'mercedes-benz', 'audi', 'porsche', 'volvo', 'jaguar'];
   638→    const economyBrands = ['ford', 'opel', 'renault', 'fiat', 'hyundai', 'kia', 'dacia'];
   639→    const premiumBrands = ['lexus', 'infiniti', 'acura'];
   640→
   641→    let baseRate = 0.08; // yıllık %8
   642→    
   643→    if (luxuryBrands.includes(make.toLowerCase())) {
   644→      baseRate = 0.12; // Lüks markalar daha hızlı değer kaybeder
   645→    } else if (economyBrands.includes(make.toLowerCase())) {
   646→      baseRate = 0.10; // Ekonomik markalar
   647→    } else if (premiumBrands.includes(make.toLowerCase())) {
   648→      baseRate = 0.09; // Premium markalar
   649→    }
   650→
   651→    // İlk 3 yıl daha hızlı değer kaybı
   652→    if (age <= 3) {
   653→      baseRate *= 1.5;
   654→    } else if (age <= 5) {
   655→      baseRate *= 1.2;
   656→    } else if (age > 10) {
   657→      baseRate *= 0.7; // Yaşlı araçlar daha yavaş değer kaybeder
   658→    }
   659→
   660→    return Math.min(baseRate, 0.35); // Maksimum %35
   661→  }
   662→
   663→  private static estimateFuelConsumption(fuelType: string, bodyType: string): number {
   664→    // 100 km'de ortalama yakıt tüketimi (litre)
   665→    const baseConsumption: Record<string, number> = {
   666→      'petrol': 7.5,
   667→      'diesel': 5.5,
   668→      'lpg': 9.5,
   669→      'electric': 0, // Elektrikli
   670→      'hybrid': 4.5,
   671→      'petrol-lpg': 9.0
   672→    };
   673→
   674→    let consumption = baseConsumption[fuelType || 'petrol'] || 7.5;
   675→
   676→    // Kasa tipine göre ayarlama
   677→    const bodyFactors: Record<string, number> = {
   678→      'sedan': 1.0,
   679→      'hatchback': 0.95,
   680→      'suv': 1.2,
   681→      'pickup': 1.3,
   682→      'coupe': 1.05,
   683→      'convertible': 1.1,
   684→      'station-wagon': 1.05,
   685→      'minivan': 1.15
   686→    };
   687→
   688→    const factor = bodyFactors[bodyType || 'sedan'] || 1.0;
   689→    return Math.round((consumption * factor) * 10) / 10;
   690→  }
   691→
   692→  private static calculateTax(year: number, engineVolume: number): number {
   693→    // 2026 MTV oranları (yaklaşık)
   694→    const currentYear = new Date().getFullYear();
   695→    const age = currentYear - year;
   696→
   697→    let taxAmount = 0;
   698→    
   699→    // Sadece 1499 cc'ye kadar olan araçlar için örnek
   700→    if (engineVolume <= 1600) {
   701→      if (age <= 3) taxAmount = 5009;
   702→      else if (age <= 6) taxAmount = 3756;
   703→      else if (age <= 11) taxAmount = 2178;
   704→      else if (age <= 15) taxAmount = 1539;
   705→      else taxAmount = 591;
   706→    } else if (engineVolume <= 2000) {
   707→      // ... 1600-2000 cc için MTV
   708→      taxAmount = 4000;
   709→    } else {
   710→      // ... 2000+ cc için MTV
   711→      taxAmount = 8000;
   712→    }
   713→
   714→    return taxAmount * 2; // Yılda iki taksit
   715→  }

   934→  }
   935→
   936→  // Sıralama
   937→  const sortField = searchParams.get('sort') || 'price';

   949→      orderBy = { mileageKm: sortOrder };

  1139→export async function getCached<T>(key: string): Promise<T | null> {
  1140→  try {
  1141→    const data = await redis.get(key);

  1148→  } catch (error) {

  1562→## 15. ÖNEMLİ UYARILAR VE EN İYİ UYGULAMALAR
  1563→
  1564→### 15.1. Scraping Etik Kuralları
  1565→
  1566→1. **Robots.txt Kontrolü:** Her adapter, hedef sitenin `robots.txt` dosyasını kontrol etmeli ve yasaklanmış alanlara istek atmamalı.
  1567→2. **Rate Limiting:** Dakikada belirli sayıda istek sınırı aşılmamalı (öneri: 10-20 istek/dakika).
  1568→3. **İş Saatleri Dışı:** Scraping işlemleri tercihen gece saatlerinde (02:00-06:00) yapılmalı.
  1569→4. **User-Agent:** Gerçek bir tarayıcı gibi görünmek için güncel User-Agent kullanılmalı.
  1570→5. **Bot Bildirimi:** Mümkünse sitelere scraping yapıldığı bildirilmeli.
  1571→
  1572→### 15.2. IP Blokajını Önleme
  1573→
  1574→1. **Rotating Proxies:** Her istek için farklı IP adresi kullanın.
  1575→2. **Residential Proxies:** Veri merkezi proxy'leri yerine residential proxy'ler tercih edin.
  1576→3. **Random Delays:** İstekler arasında random gecikmeler ekleyin.
  1577→4. **Session Rotation:** Cookie ve session'ları düzenli olarak yenileyin.
  1578→5. **User-Agent Rotation:** Her istekte farklı User-Agent kullanın.
  1579→
  1580→### 15.3. Veri Kalitesi
  1581→
  1582→1. **Validation:** Her ilan verisi doğrulanmalı (fiyat > 0, yıl geçerli, vb.).
  1583→2. **Dedup:** Aynı ilan birden fazla kez toplanmamalı.
  1584→3. **Normalization:** Tüm veriler ortak formata dönüştürülmeli.
  1585→4. **Freshness:** Eski ilanlar düzenli olarak temizlenmeli.
  1586→5. **Data Completeness:** Eksik veriler tespit edilmeli ve raporlanmalı.
  1587→
  1588→### 15.4. Performans
  1589→
  1590→1. **Bulk Operations:** Veritabanı işlemleri batch halinde yapılmalı.
  1591→2. **Connection Pooling:** Veritabanı ve Redis bağlantıları havuzlanmalı.
  1592→3. **Async/Await:** Tüm I/O işlemleri asenkron yapılmalı.
  1593→4. **Caching:** Sık kullanılan veriler cache'lenmeli.
  1594→5. **Indexing:** Sorgular için uygun indeksler oluşturulmalı.

  1596→### 15.5. Güvenlik

  1598→1. **Token Koruma:** Admin endpoint'leri token ile korunmalı.
