   370→  isDateWithinRange(dateStr) {
   371→    if (!CONFIG.maxDaysOld) return true;
   372→    
   373→    try {
   374→      // Türkçe tarih formatı: "19 Haziran 2026"
   375→      const months = {
   376→        'Ocak': 0, 'Şubat': 1, 'Mart': 2, 'Nisan': 3, 'Mayıs': 4, 'Haziran': 5,
   377→        'Temmuz': 6, 'Ağustos': 7, 'Eylül': 8, 'Ekim': 9, 'Kasım': 10, 'Aralık': 11
   378→      };
   379→      
   380→      const parts = dateStr.trim().split(' ');
   381→      if (parts.length === 3) {
   382→        const day = parseInt(parts[0]);
   383→        const month = months[parts[1]];
   384→        const year = parseInt(parts[2]);
   385→        
   386→        if (!isNaN(day) && month !== undefined && !isNaN(year)) {
   387→          const date = new Date(year, month, day);
   388→          const now = new Date();
   389→          const diffDays = (now - date) / (1000 * 60 * 60 * 24);
   390→          return diffDays <= CONFIG.maxDaysOld;
   391→        }
   392→      }
   393→    } catch (e) {
   394→      // Tarih parse edilemezse kabul et
   395→      return true;
   396→    }
   397→    
   398→    return true;
   399→  }

   486→              const el_ = el.querySelector(sel);
   487→              if (el_) { year = parseInt(el_.textContent.trim()) || 0; break; }
   488→            }

   497→              }

   572→                mileage_km: km > 0 ? km : undefined,
   573→                fuel_type: fuel || undefined,
   574→                transmission: transmission || undefined,
   575→                city: city || undefined,
   576→                image_url: imageUrl || undefined,
   577→                date_text: dateStr || '',
   578→                scraped_at: new Date().toISOString(),
   579→                _index: index + 1
   580→              });
   581→            }
   582→
   583→          } catch (error) {
   584→            console.log(`⚠️ İlan ${index + 1} parse hatası:`, error.message);
   585→          }
   586→        });
   587→
   588→        return items;
   589→      }, SELECTORS.search);
   590→
   591→      console.log(`✅ ${pageListings.length} ilan bulundu`);
   592→      return pageListings;
   593→
   594→    } catch (error) {
   595→      console.log(`❌ Sayfa ${pageNumber} hatası:`, error.message);
   596→      this.errors.push({ page: pageNumber, url, error: error.message });
   597→      return [];
   598→    }

   601→  // ================================================================
   602→  // 3.9. DETAY SAYFASI SCRAPING

   605→  async scrapeDetail(listingItem) {
   606→    const url = listingItem.source_url;
   607→    
   608→    // Daha önce işlendi mi?
   609→    if (this.processedUrls.has(url)) {
   610→      console.log(`   ⏭️  Zaten işlendi: ${url}`);
   611→      return null;
   612→    }
   613→    this.processedUrls.add(url);
   614→
   615→    console.log(`\n🔍 Detay: ${url}`);
   616→
   617→    try {
   618→      // Detay sayfasına git
   619→      await this.page.goto(url, {
   620→        waitUntil: 'networkidle',
   621→        timeout: 45000
   622→      });
   623→
   624→      await this.simulateHumanBehavior();
   625→
   626→      // İlan aktif mi kontrol et
   627→      const isActive = await this.checkIfActive();
   628→      
   629→      if (!isActive) {
   630→        console.log(`   ⏭️  İlan yayından kalkmış veya satılmış`);
   631→        this.skipped.push({ url, reason: 'inactive' });
   632→        return null;
   633→      }
   634→
   635→      // Detay verilerini çek
   636→      const detail = await this.page.evaluate((selectors) => {
   637→        const data = {};
   638→
   639→        // Canonical URL
   640→        const canonicalEl = document.querySelector(selectors.canonical);
   641→        if (canonicalEl) {
   642→          data.canonical_url = canonicalEl.getAttribute('href') || '';
   643→        }
   644→
   645→        // Başlık
   646→        for (const sel of selectors.title) {
   647→          const el = document.querySelector(sel);
   648→          if (el) { data.title = el.textContent.trim(); break; }
   649→        }
   650→
   651→        // Fiyat
   652→        const priceEl = document.querySelector(selectors.price);
   653→        if (priceEl) {
   654→          const text = priceEl.textContent.trim();
   655→          const cleaned = text.replace(/[^0-9,.]/g, '').replace(/\./g, '').replace(/,/g, '.');
   656→          data.price = parseFloat(cleaned) || 0;
   657→        }
   658→
   659→        // İlan No
   660→        const idEl = document.querySelector(selectors.listingId);
   661→        if (idEl) { data.id = idEl.textContent.trim(); }
   662→
   663→        // Konum
   664→        for (const sel of selectors.location) {
   665→          const el = document.querySelector(sel);
   666→          if (el) { data.location = el.textContent.trim(); break; }
   667→        }
   668→
   669→        // Satıcı
   670→        const sellerContainer = document.querySelector(selectors.seller.container);
   671→        if (sellerContainer) {
   672→          const nameEl = sellerContainer.querySelector(selectors.seller.name);
   673→          data.seller_name = nameEl ? nameEl.textContent.trim() : '';
   674→          
   675→          const memberTypeEl = sellerContainer.querySelector(selectors.seller.memberType);
   676→          data.seller_member_type = memberTypeEl ? memberTypeEl.textContent.trim() : '';
   677→          
   678→          const badgeEl = sellerContainer.querySelector(selectors.seller.badge);
   679→          data.seller_badge = badgeEl ? badgeEl.textContent.trim() : '';
   680→          
   681→          const logoEl = sellerContainer.querySelector(selectors.seller.logo);
   682→          data.seller_logo = logoEl ? logoEl.getAttribute('src') || '' : '';
   683→        }
   684→
   685→        // Firma
   686→        const firmOwnerEl = document.querySelector(selectors.firm.owner);
   687→        data.firm_owner = firmOwnerEl ? firmOwnerEl.textContent.trim() : '';
   688→        
   689→        const docEl = document.querySelector(selectors.firm.documentNumber);
   690→        data.document_number = docEl ? docEl.textContent.trim() : '';
   691→
   692→        // Açıklama
   693→        let description = '';
   694→        const descEls = document.querySelectorAll(selectors.description);
   695→        descEls.forEach(el => {
   696→          description += el.textContent.trim() + '\n';
   697→        });
   698→        data.description = description.trim();
   699→
   700→        // Görseller
   701→        data.images = [];
   702→        const imgEls = document.querySelectorAll(selectors.images);
   703→        imgEls.forEach(img => {
   704→          const src = img.getAttribute('src') || 
   705→                      img.getAttribute('data-src') || 
   706→                      img.getAttribute('srcset')?.split(' ')[0];
   707→          if (src && src.includes('mncdn.com')) {
   708→            data.images.push(src);
   709→          }
   710→        });
   711→
   712→        // Teknik Detaylar
   713→        data.technical_details = {};
   714→        const carInfoContainer = document.querySelector(selectors.carInformation.container);
   715→        if (carInfoContainer) {
   716→          const sections = carInfoContainer.querySelectorAll(selectors.carInformation.sections);
   717→          sections.forEach(section => {
   718→            const titleEl = section.querySelector(selectors.carInformation.sectionTitle);
   719→            const sectionTitle = titleEl ? titleEl.textContent.trim() : 'Genel';
   720→            
   721→            data.technical_details[sectionTitle] = {};
   722→            const items = section.querySelectorAll(selectors.carInformation.items);
   723→            items.forEach(item => {
   724→              const keyEl = item.querySelector(selectors.carInformation.key);
   725→              const valueEl = item.querySelector(selectors.carInformation.value);
   726→              if (keyEl && valueEl) {
   727→                const key = keyEl.textContent.trim();
   728→                const value = valueEl.textContent.trim();
   729→                data.technical_details[sectionTitle][key] = value;
   730→              }
   731→            });
   732→          });
   733→        }
   734→
   735→        // Özellikler
   736→        data.properties = {};
   737→        const propsContainer = document.querySelector(selectors.properties.container);
   738→        if (propsContainer) {
   739→          const items = propsContainer.querySelectorAll(selectors.properties.items);
   740→          items.forEach(item => {
   741→            const keyEl = item.querySelector(selectors.properties.key);
   742→            const valueEl = item.querySelector(selectors.properties.value);
   743→            if (keyEl && valueEl) {
   744→              const key = keyEl.textContent.trim();
   745→              const value = valueEl.textContent.trim();
   746→              data.properties[key] = value;
   747→            }
   748→          });
   749→        }
   750→
   751→        return data;
   752→      }, SELECTORS.detail);
   753→
   754→      if (!detail || !detail.id) {
   755→        console.log(`   ❌ Detay verisi çekilemedi`);
   756→        this.skipped.push({ url, reason: 'no_data' });
   757→        return null;
   758→      }
   759→
   760→      // ---- VERİLERİ BİRLEŞTİR ----
   761→      const merged = {
   762→        ...listingItem,
   763→        ...detail,
   764→        // Arama sonuçlarındaki veriler detaydan gelenlerle güncellenir
   765→        source_url: detail.canonical_url || listingItem.source_url,
   766→        title: detail.title || listingItem.title,
   767→        price: detail.price || listingItem.price,
   768→        year: detail.year || listingItem.year,
   769→        mileage_km: detail.mileage_km || listingItem.mileage_km,
   770→        fuel_type: detail.fuel_type || listingItem.fuel_type,
   771→        transmission: detail.transmission || listingItem.transmission,
   772→        city: detail.city || listingItem.city,
   773→        make: detail.make || listingItem.make,
   774→        model: detail.model || listingItem.model,
   775→      };
   776→
   777→      // Properties'ten ek bilgiler
   778→      if (detail.properties) {
   779→        if (!merged.year) merged.year = parseInt(detail.properties['Yıl']) || 0;
   780→        if (!merged.mileage_km) {
   781→          merged.mileage_km = parseInt(detail.properties['Kilometre']?.replace(/[^0-9]/g, '')) || 0;
   782→        }
   783→        if (!merged.fuel_type) merged.fuel_type = detail.properties['Yakıt Tipi'] || '';
   784→        if (!merged.transmission) merged.transmission = detail.properties['Vites Tipi'] || '';
   785→        if (!merged.body_type) merged.body_type = detail.properties['Kasa Tipi'] || '';
   786→        if (!merged.color) merged.color = detail.properties['Renk'] || '';
   787→        if (!merged.city) merged.city = detail.properties['İl'] || detail.location || '';
   788→        if (!merged.seller_type) merged.seller_type = detail.properties['Kimden'] || '';
   789→      }
   790→
   791→      console.log(`   ✅ ${merged.make} ${merged.model} - ${merged.price.toLocaleString()} TL`);
   792→      return merged;
   793→
   794→    } catch (error) {
   795→      console.log(`   ❌ Detay hatası:`, error.message);
   796→      this.errors.push({ url, error: error.message });
   797→      return null;
   798→    }
   799→  }

   805→  async scrape() {
   806→    try {
   807→      console.log('═══════════════════════════════════════════════════');
   808→      console.log('🚗 ARABAM.COM SCRAPER v3.2 - GÜNCEL İLANLAR');
   809→      console.log('═══════════════════════════════════════════════════');
   810→      console.log(`📄 Max sayfa: ${CONFIG.maxPages}`);

   812→      console.log(`🖥️  Görünür mod: ${!CONFIG.headless}`);
   813→      console.log('═══════════════════════════════════════════════════\n');
   814→
   815→      await this.init();
   816→
   817→      // ---- Tekil ilan detayı ----
   818→      if (CONFIG.detailId) {
   819→        const fakeItem = { 
   820→          id: CONFIG.detailId, 
   821→          source_url: `https://www.arabam.com/ilan/detay/${CONFIG.detailId}` 
   822→        };
   823→        const detail = await this.scrapeDetail(fakeItem);
   824→        if (detail) {
   825→          this.listings.push(detail);
   826→          this.saveResults();
   827→        }
   828→        return;
   829→      }
   830→
   831→      // ---- Sayfaları tara ----
   832→      for (let pageNum = 1; pageNum <= CONFIG.maxPages; pageNum++) {
   833→        if (this.listings.length >= CONFIG.maxTotalListings) {
   834→          console.log(`\n✅ ${CONFIG.maxTotalListings} ilan hedefine ulaşıldı.`);
   835→          break;
   836→        }
   837→
   838→        const pageListings = await this.scrapeSearch(pageNum);
   839→        this.totalFound += pageListings.length;
   840→
   841→        if (pageListings.length === 0) {
   842→          console.log(`\n⚠️ Sayfa ${pageNum}'da ilan bulunamadı, durduruluyor.`);
   843→          break;
   844→        }
   845→
   846→        // Her ilan için detay çek
   847→        let processed = 0;
   848→        for (const item of pageListings) {
   849→          if (this.listings.length >= CONFIG.maxTotalListings) break;
   850→          
   851→          const detail = await this.scrapeDetail(item);
   852→          processed++;
   853→          
   854→          if (detail) {
   855→            this.listings.push(detail);
   856→            console.log(`   📦 Toplam: ${this.listings.length}/${CONFIG.maxTotalListings}`);
   857→          }
   858→          
   859→          // İşlemler arası bekle
   860→          if (processed < pageListings.length) {
   861→            const waitTime = this.getRandomDelay(1500, 3000);
   862→            await this.page.waitForTimeout(waitTime);
   863→          }
   864→        }
   865→
   866→        console.log(`\n📊 Sayfa ${pageNum} tamamlandı: ${this.listings.length} aktif ilan`);
   867→
   868→        if (pageNum < CONFIG.maxPages && this.listings.length < CONFIG.maxTotalListings) {
   869→          console.log(`⏳ Sayfalar arası ${CONFIG.delays.betweenPages}ms bekleniyor...`);
   870→          await this.page.waitForTimeout(CONFIG.delays.betweenPages);
   871→        }
   872→      }
   873→
   874→      // ---- Sonuçları kaydet ----
   875→      this.saveResults();
   876→
   877→      // ---- Özet rapor ----
   878→      console.log('\n📊 SCRAPING ÖZETİ:');
   879→      console.log(`   ✅ Başarılı: ${this.listings.length} ilan`);
   880→      console.log(`   ⏭️  Atlanan: ${this.skipped.length} ilan (satılmış/kaldırılmış)`);

   882→      console.log(`   📄 Taranan sayfa: ${CONFIG.maxPages}`);
   883→      console.log(`   🔍 Toplam bulunan: ${this.totalFound} ilan`);
   884→
   885→    } catch (error) {
   886→      console.error('\n❌ Scraping hatası:', error);
   887→    } finally {
   888→      if (this.browser) {
   889→        await this.browser.close();
   890→        console.log('\n🔒 Browser kapatıldı');
   891→      }
   892→    }
   893→  }

   899→  saveResults() {
   900→    const duration = (Date.now() - this.startTime) / 1000;
   901→
   902→    const output = {
   903→      metadata: {
   904→        site: 'arabam.com',
   905→        scraped_at: new Date().toISOString(),
   906→        total_listings: this.listings.length,
   907→        total_pages: CONFIG.maxPages,
   908→        duration_seconds: Math.round(duration),
   909→        only_active: CONFIG.onlyActive,
   910→        max_days_old: CONFIG.maxDaysOld || 'all',
   911→        config: {
   912→          max_pages: CONFIG.maxPages,
   913→          max_total_listings: CONFIG.maxTotalListings,
   914→          headless: CONFIG.headless,
   915→        }
   916→      },
   917→      stats: {
   918→        total_found: this.totalFound,
   919→        active: this.listings.length,
   920→        skipped: this.skipped.length,
   921→        errors: this.errors.length
   922→      },
   923→      skipped: this.skipped.slice(0, 50),
   924→      errors: this.errors.slice(0, 50),
   925→      listings: this.listings
   926→    };
   927→
   928→    const outputPath = path.join(__dirname, 'arabam_data.json');
   929→    fs.writeFileSync(outputPath, JSON.stringify(output, null, 2));
   930→
   931→    console.log('\n═══════════════════════════════════════════════════');
   932→    console.log('✅ VERİLER KAYDEDİLDİ!');
   933→    console.log(`📁 Dosya: ${outputPath}`);
   934→    console.log(`📊 Toplam aktif ilan: ${this.listings.length}`);
   935→    console.log(`⏱️  Süre: ${Math.round(duration)} saniye`);
   936→    console.log('═══════════════════════════════════════════════════');
   937→
   938→    if (this.listings.length > 0) {
   939→      const validPrices = this.listings.filter(l => l.price > 0);
   940→      if (validPrices.length > 0) {
   941→        const prices = validPrices.map(l => l.price);
   942→        const avg = Math.round(prices.reduce((a, b) => a + b, 0) / prices.length);
   943→        const min = Math.min(...prices);
   944→        const max = Math.max(...prices);
   945→        console.log('\n💰 Fiyat İstatistikleri:');
   946→        console.log(`   Min: ${min.toLocaleString()} TL`);
   947→        console.log(`   Max: ${max.toLocaleString()} TL`);
   948→        console.log(`   Ort: ${avg.toLocaleString()} TL`);
   949→        console.log(`   Toplam: ${prices.length} ilan`);
   950→      }
   951→
   952→      const makes = {};
   953→      this.listings.forEach(l => {
   954→        const m = l.make || 'Bilinmiyor';
   955→        makes[m] = (makes[m] || 0) + 1;
   956→      });
   957→      const sortedMakes = Object.entries(makes)
   958→        .sort((a, b) => b[1] - a[1])
   959→        .slice(0, 5);
   960→
   961→      if (sortedMakes.length > 0) {
   962→        console.log('\n🏷️  En Çok Görülen Markalar:');
   963→        sortedMakes.forEach(([make, count]) => {
   964→          const pct = Math.round((count / this.listings.length) * 100);
   965→          console.log(`   ${make}: ${count} ilan (${pct}%)`);
   966→        });
   967→      }
   968→
   969→      console.log('\n📊 Örnek Veri (ilk 3 ilan):');
   970→      this.listings.slice(0, 3).forEach((l, i) => {
   971→        console.log(`   ${i + 1}. ${l.make} ${l.model} - ${l.price.toLocaleString()} TL`);
   972→      });
   973→    }
   974→
   975→    console.log('\n═══════════════════════════════════════════════════');
   976→  }
   977→}

  1247→class DateParser {
  1248→  constructor() {
  1249→    this.months = {
  1250→      'Ocak': 0, 'Şubat': 1, 'Mart': 2, 'Nisan': 3, 'Mayıs': 4, 'Haziran': 5,
  1251→      'Temmuz': 6, 'Ağustos': 7, 'Eylül': 8, 'Ekim': 9, 'Kasım': 10, 'Aralık': 11
  1252→    };
  1253→  }
  1254→
  1255→  // Türkçe tarih string'ini Date objesine çevir
  1256→  parseTurkishDate(dateStr) {
  1257→    if (!dateStr) return null;
  1258→    
  1259→    try {
  1260→      // "19 Haziran 2026" formatı
  1261→      const parts = dateStr.trim().split(' ');
  1262→      if (parts.length === 3) {
  1263→        const day = parseInt(parts[0]);
  1264→        const month = this.months[parts[1]];
  1265→        const year = parseInt(parts[2]);
  1266→        
  1267→        if (!isNaN(day) && month !== undefined && !isNaN(year)) {
  1268→          return new Date(year, month, day);
  1269→        }
  1270→      }
  1271→      
  1272→      // "19.06.2026" formatı
  1273→      if (dateStr.includes('.')) {
  1274→        const parts = dateStr.split('.');
  1275→        if (parts.length === 3) {
  1276→          const day = parseInt(parts[0]);
  1277→          const month = parseInt(parts[1]) - 1;
  1278→          const year = parseInt(parts[2]);
  1279→          if (!isNaN(day) && !isNaN(month) && !isNaN(year)) {
  1280→            return new Date(year, month, day);
  1281→          }
  1282→        }
  1283→      }
  1284→      
  1285→      // "2026-06-19" formatı
  1286→      if (dateStr.includes('-')) {
  1287→        const parts = dateStr.split('-');
  1288→        if (parts.length === 3) {
  1289→          const year = parseInt(parts[0]);
  1290→          const month = parseInt(parts[1]) - 1;
  1291→          const day = parseInt(parts[2]);
  1292→          if (!isNaN(day) && !isNaN(month) && !isNaN(year)) {
  1293→            return new Date(year, month, day);
  1294→          }
  1295→        }
  1296→      }
  1297→      
  1298→      return null;
  1299→    } catch (e) {
  1300→      return null;
  1301→    }
  1302→  }
  1303→
  1304→  // Tarihin belirtilen gün içinde olup olmadığını kontrol et
  1305→  isWithinDays(date, maxDays) {
  1306→    if (!date) return true; // Tarih yoksa kabul et
  1307→    if (!maxDays) return true; // Filtre yoksa kabul et
  1308→    
  1309→    const now = new Date();
  1310→    const diffDays = (now - date) / (1000 * 60 * 60 * 24);
  1311→    return diffDays <= maxDays;
  1312→  }
  1313→
  1314→  // Tarih string'ini okunabilir formata çevir
  1315→  formatDate(date) {
  1316→    if (!date) return 'Tarih yok';
  1317→    return date.toLocaleDateString('tr-TR', {
  1318→      day: 'numeric',
  1319→      month: 'long',
  1320→      year: 'numeric'
  1321→    });
  1322→  }
